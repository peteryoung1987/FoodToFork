Setting up sheet write-back

Follow these steps once — takes about 3 minutes:

  1. Open your Google Sheet
  2. Click Extensions → Apps Script
  3. Delete any existing code and paste the script below
  4. Click Deploy → New deployment
  5. Type: Web app, Execute as: Me, Who has access: Anyone
  6. Click Deploy, copy the URL and paste it above
function doPost(e) { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); if (data.action === 'writeIngredients') { const sheet = ss.getSheetByName('Ingredients') || ss.getActiveSheet(); const headers = ['Item','Qty','Storage','Category','Use By','Used']; sheet.clearContents(); sheet.getRange(1,1,1,headers.length).setValues([headers]); if (data.rows.length > 0) { sheet.getRange(2,1,data.rows.length,headers.length).setValues(data.rows); } } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
